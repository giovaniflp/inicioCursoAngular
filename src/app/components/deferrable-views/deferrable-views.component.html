Deferrable Views

Defer com ON

@defer (on interaction(Ativador)) {
Defer Div
} @placeholder {
Placeholder...
} @loading {
Carregando...
} @error {
Erro de carregamento!
}

Defer com WHEN

@defer (when condition) {
When Div
} @placeholder(minimum 5s) {
When placeholder...
} @loading(after 100ms; minimum 5s) {
When Carregando...
} @error {
Erro
}
